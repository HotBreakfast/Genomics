Option Explicit
Dim data As String
Private Sub CountingWords()
Dim r As Long
'turn off alerts
Application.DisplayAlerts = False
On Error Resume Next
'delete other worksheets
Dim ws As Worksheet
For Each ws In ThisWorkbook.Worksheets
    If ws.Name <> "Sheet1" Then ws.Delete
Next
'turn on alerts
Application.DisplayAlerts = True
'clear cells for import
Cells.Clear
Range("A1").Activate

Dim Pattern, text As String
Pattern = InputBox("What is the Nucleotide Pattern?", "Enter Pattern")
Dim dataFile As String
dataFile = InputBox("Enter Genome Text File :", "Please enter the file path of the Genome you want to evaluate ", "C:\Users\Allen\Desktop\data.txt")
'OPEN TEXT FILE WITH GENOME SEQUENCE AND IMPORT LINE BY LINE DOWN COLUMN A
Open dataFile For Input As #1
r = 0
Do Until EOF(1)
    Line Input #1, data
    If r < 1048576 Then
        ActiveCell.Offset(r, 0) = data
        text = ActiveCell.Offset(r, 0).Value
        r = r + 1
        Call CountLongWords(text, Pattern)
    ElseIf r > 1048576 Then
        MsgBox "File is too big at this time"
        Exit Sub
    End If
Loop
Close #1
Columns("A:A").Delete shift:=xlToLeft
End Sub
Private Function CountLongWords(text, Pattern)
Dim i, k, count As Long
k = Len(text)

Cells(1, 2) = ("The count for " & Pattern & " is :")
For i = 1 To k
    Dim temp As String
    temp = Mid(text, i, Len(Pattern))
    If temp = Pattern Then
        Cells(1, 3).Value = Cells(1, 3).Value + 1
    End If
Next i
Columns.AutoFit
End Function
